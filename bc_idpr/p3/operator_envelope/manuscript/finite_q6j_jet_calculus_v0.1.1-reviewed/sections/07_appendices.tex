\section{First three generator derivatives}
Direct differentiation gives
\begin{align}
K&=F'F^T,\\
K'&=F''F^T+F'F'^T,\\
K''&=F'''F^T+2F''F'^T+F'F''^T.
\end{align}
These agree with Theorem \ref{thm:generatorrecurrence} after conversion between normalized Taylor coefficients and ordinary derivatives.

\section{Exact carrier and certificate pseudocode}
\label{app:pseudocode}
\paragraph{Algebraic carrier enumeration.}
\begin{verbatim}
for J in {1,...,6}^4:
    require even sum(J)
    compute the two exact integer channel sets E(J), G(J)
    require |E(J)| = |G(J)| = 2
    for every e in E(J), f in G(J):
        construct exact integer Racah limits A_i, B_j
        require max(A_i) <= min(B_j)
        require all triangle, numerator, denominator,
                and dimension indices lie in {0,...,10}
    retain J
\end{verbatim}
This returns 283 ordered carriers and 24 sorted families.

\paragraph{Arb Taylor--Cauchy certificate.}
\begin{verbatim}
fix N=50, Rout=pi/120, Rcert=pi/1200
for precision in {192,256} bits:
    for each canonical external-label family:
        compute omega_0,...,omega_50 as Arb balls
        compute an Arb outer-circle majorant M
        lower = abs_lower(omega_0)
                - sum(abs_upper(omega_k) Rcert^k, k=1..50)
                - abs_upper(M) (Rcert/Rout)^51/(1-Rcert/Rout)
        require lower > 0
require identical worst family at both precisions
\end{verbatim}
